 ----- Работа с файловой системой ------------------------------------
 cd				- скажи мне путь, куда идём
 cd ..				- назад
 pwd				- где я?
 ls				- а что в текущей директории?
 ls -a				- и скрытые покажи!
 ls -l				- развёрнуто покажи:
					1. информацию о правах доступа
					2. числа ссылок на каждый элемент файловой системы 
					3,4 список владельцев и групп
					5. размер объекта
					6. время последнего изменения (mtime) объекта
					7. имена
 ls -R				- всё что вложено покажи рекурсивно
 ls -i				- инод маркерует директории и файлы

 mkdir				- создать директрию
 mkdir -p			- и все поддиректории
 rmdir				- удалить директорию
 rm -rf				- удалить директорию и всё, что есть внутри

 # дать права gf
 sudo chmod -R 777 /glassfish5

 ----- Работа с файлами ----------------------------------------------
 touch				- если нет - создать, есть обновить
 echo "text" > filename		- записать вывод в file
 cat filename			- вывести содержимое file на консоль
 cp file1 file2			- копировать f1 в f2
 mv file1 file2			- переименовать f1 в f2
 mv /dir/f1.txt /tmp		- переместить из f1 в tmp
 rm 				- удалить навсегда
 rm -i				- спросить и удалить
 tar xvfz *.tar.gz		- разархивировать tar.gz

 ----- поиск ---------------------------------------------------------
 which ls 			- есть ли программа ls в PATH
 find /usr -name README 	- поиск
 find      -iname		- поиск без учёта регистра
 find . -name \? -mtime -1 	- найти изменённые за последние 24 ч.
 locate bin/ls			- поиск совпадений пути
 slocate			- secure locate 
 updatedb			- если locate не прошел

 # работа с найденными файлами -exec найденное сюда '{}' завершить ';'
 find /usr/bin -type f -size -50c -exec ls -l '{}' ';'

 ----- джокеры -------------------------------------------------------
 rm file[1-8]			- удалить файлы по указанной маске
 rm file[12]
 rm [Ff]ile
 rm file[!9]			- все кроме file9
 rm file*			- любое окончание, или его отсутствие
 rm file?			- один символ в окончании
 rm [A-Za-z]* 			- все файлы, которые начинаются с латинской буквы

 ----- управление процессами -----------------------------------------
 xeyes -center red		- проверяем можем ли мы работать с процессами
			 	  Ctrl+c убивает этот процесс а Ctrl+z стопит 
 bg и fg			- "back.. ground" "foreg..."
 xeyes -center blue &		- '&' запустить в фоне
 jobs -l			- процессы запущенных в этой сессии bash
 ps ax				- все процессы системы
 ps x --forest			- в виде дерева
 top				- постоянно обновляющийся список процессов
 kill 14023			- убивает процесс по id
 kill -s SIGKILL 16224		- дубинка по-больше

 sudo dmidecode | less		- информация о системе
 cat /etc/*release*		- информация об ОС

 deb				- пакет
